## Request

Create a new booking for a property. Guest information can be provided either by referencing an existing guest ID or by providing guest details directly.

<ParamField body name="property_id" type="integer" required>
  ID of the property being booked
</ParamField>

<ParamField body name="check_in_date" type="string" format="date" required>
  Check-in date (YYYY-MM-DD)
</ParamField>

<ParamField body name="check_out_date" type="string" format="date" required>
  Check-out date (YYYY-MM-DD)
</ParamField>

<ParamField body name="guests_count" type="integer" required>
  Number of guests
</ParamField>

<ParamField body name="nightly_rate" type="number" required>
  Nightly rate for this booking
</ParamField>

<ParamField body name="guest_id" type="integer">
  ID of existing guest (if guest already exists)
</ParamField>

<ParamField body name="guest_first_name" type="string">
  Guest's first name (if creating new guest)
</ParamField>

<ParamField body name="guest_last_name" type="string">
  Guest's last name (if creating new guest)
</ParamField>

<ParamField body name="guest_email" type="string" format="email">
  Guest's email address (if creating new guest)
</ParamField>

<ParamField body name="guest_phone" type="string">
  Guest's phone number
</ParamField>

<ParamField body name="cleaning_fee" type="number">
  Cleaning fee amount
</ParamField>

<ParamField body name="service_fee" type="number">
  Service fee amount
</ParamField>

<ParamField body name="taxes" type="number">
  Tax amount
</ParamField>

<ParamField body name="source" type="string">
  Booking source (e.g., "direct", "airbnb", "booking.com")
</ParamField>

<ParamField body name="special_requests" type="string">
  Special requests or notes
</ParamField>

        'taxes': 100.00,

booking = response.json()
